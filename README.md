# DetectOS-BrowserInPHP
Detect which user browser and operating system is use using PHP Generated by AI 

```PHP

<?php 

function getUserInfo() {
    // Check if HTTP_USER_AGENT is set
    if (!isset($_SERVER['HTTP_USER_AGENT'])) {
        return [
            'os' => 'User Agent not available',
            'browser' => 'User Agent not available'
        ];
    }

    $userAgent = $_SERVER['HTTP_USER_AGENT'];
    
    // List of operating systems
    $osList = [
        'Windows',
        'macOS',
        'Ubuntu',
        'Fedora',
        'Debian',
        'CentOS',
        'Arch Linux',
        'Mint',
        'Red Hat Enterprise Linux',
        'openSUSE',
        'Chrome OS',
        'FreeBSD',
        'Solaris',
        'Android',
        'iOS',
        'Windows Phone',
        'KaiOS',
        'HarmonyOS',
        'Windows Server',
        'Ubuntu Server',
        'AIX',
        'HP-UX',
        'Embedded Linux',
        'FreeRTOS',
        'VxWorks',
        'QNX',
        'OpenVMS',
        'z/OS',
        'BeOS'
    ];

    // List of browsers
    $browserList = [
        'Edg', // Microsoft Edge
        'Chrome',
        'Firefox',
        'Safari',
        'Edge',
        'Opera',
        'MSIE', // Internet Explorer
        'Trident' // IE 11
    ];

    // Determine the OS
    $os = 'Unknown OS';
    foreach ($osList as $osName) {
        if (stripos($userAgent, $osName) !== false) {
            $os = $osName; // Found the OS
            break;
        }
    }

    // Determine the browser
    $browser = 'Unknown Browser';
    foreach ($browserList as $browserName) {
        if (stripos($userAgent, $browserName) !== false) {
            $browser = $browserName; // Found the browser
            break;
        }
    }

    return [
        'os' => $os,
        'browser' => $browser
    ];
}

// Usage
$userInfo = getUserInfo();
echo "The user is visiting from: " . $userInfo['os'] . " using " . $userInfo['browser'] . ".";


```
